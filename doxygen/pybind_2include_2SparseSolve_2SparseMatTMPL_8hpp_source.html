427/*//=======================================================
428// ● 一時配列を確定させる
429//=======================================================*/
434template<typename DType>
435void SparseMatTMPL<DType>::fix(bool toSquare){
436 /* スパース位置が確定していたらなにもしない*/
437 if(is_fix){
438 std::cout << "already fixed sparse!"<< std::endl;
439 return;
440 }
441 /* Eigenに代入 */
442 slv_int max_retu = 0;
443 std::vector<Eigen::Triplet<DType>> tripletList;
444 for(slv_int i = 0; i < size; i++) {
445 const slv_int ss = tempMat[i].size();
446 if(ss == 0) {
447 continue;
448 }
449 for(auto& itr : tempMat[i]) {
450 const slv_int pos = itr.first;
451 const DType val = itr.second;
452 tripletList.push_back(Eigen::Triplet<DType>(i, pos, val));
453 if(pos > max_retu){
454 max_retu = pos;
455 }
456 }
457 std::map<slv_int, DType> empty; empty.clear();
458 tempMat[i].swap(empty);
459 }
460 max_retu++;
461 /* Eigenに確定(列方向の最大値も指定する必要あり)。 */
462 /* toSquare=true なら、行数の方が大きいなら正方サイズにする */
463 /* toSquare=false なら、長方形のまま */
464 if(toSquare){
465 if(max_retu < size){
466 tripletList.push_back(Eigen::Triplet<DType>(size-1, size-1, 0.0));
467 max_retu = size;
468 }
469 }
470 matrix = Eigen::SparseMatrix<DType, Eigen::RowMajor>(size, max_retu);
471 matrix.setFromTriplets(tripletList.begin(), tripletList.end());
472
473 tempMat.reset();
474 is_fix = true;
475}

635template<typename DType>
636slv_int SparseMatTMPL<DType>::getMaxCol()const{

649 }

652 }

724/*//=======================================================
725// ● フラグがTrueの位置の列・行のデータを削除（FEM用）
726//=======================================================*/
731template<typename DType>
732void SparseMatTMPL<DType>::delFlagPosition(const bool* flag){
733 if(!is_fix) return;
734
735 /* 行列位置を取得 */
736 slv_int* start_pos1 = new slv_int[size];
737 slv_int* end_pos1 = new slv_int[size];
738 this->getCols(start_pos1, end_pos1);
739 auto col_ptr1 = matrix.innerIndexPtr();
740 auto val_ptr1 = matrix.valuePtr();
741
742 /* 削除分のID再計算 */
743 std::map<slv_int, slv_int> id_map;
744 slv_int counter=0;
745 for(slv_int i = 0; i < size; i++){
746 if(flag[i]){
747 continue;
748 }
749 id_map[i] = counter;
750 counter++;
751 }
752
753 slv_int max_retu = 0;
754 std::vector<Eigen::Triplet<DType>> tripletList;
755 /* 再度探索 */
756 for(slv_int i = 0 ; i < size ; i++){
757 /* フラグがonなら無視 */
758 if(flag[i]){
759 continue;
760 }
761 for(slv_int j = start_pos1[i] ; j < end_pos1[i] ; j++){
762 slv_int col = col_ptr1[j];
763 /* フラグがonなら無視 */
764 if(flag[col]){
765 continue;
766 }
767 /* フラグがoffなら加える */
768 DType val = val_ptr1[j];
769 tripletList.push_back(Eigen::Triplet<DType>(id_map[i], id_map[col], val));
770 if(max_retu < id_map[col]){
771 max_retu = id_map[col];
772 }
773 }
774 }
775 delete[] start_pos1;
776 delete[] end_pos1;
777 /* 再セット */
778 max_retu++;
779 size = counter;
780 matrix = Eigen::SparseMatrix<DType, Eigen::RowMajor>(counter, max_retu);
781 matrix.setFromTriplets(tripletList.begin(), tripletList.end());
782}
